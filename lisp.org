** Append list manually
#+begin_src lisp
  (defun my-append (a b)
    (if (null a)
        b
        (cons (car a)
  	    (my-append (cdr a) b))))

  (my-append '(1 2) '(3 4))
#+end_src
- first list copy hoti hai
- second list append hoti hai
- Time Complexity: O(n)

** Reverse list manually

*** Simple Recursive Approach
#+begin_src lisp
  (defun my-reverse (lst)
    (if (null lst)
        nil
        (append (my-reverse (cdr lst))
  	      (list (car lst)))))

  (my-reverse '(1 2 3))
#+end_src
Steps:
- (1 2 3)
- → reverse (2 3) + (1)
- → reverse (3) + (2) + (1)
- → (3) + (2) + (1)
- → (3 2 1)

*Drawback*:
- *append* har step pe expensive hai → *O(n²)*


*** Tail Recursive (Efficient Way)
#+begin_src lisp
  (defun my-reverse-tail (lst &optional (acc nil))
    (if (null lst)
        acc
        (my-reverse-tail (cdr lst)
  		       (cons (car lst) acc))))

  (my-reverse-tail '(1 2 3))
#+end_src

| *Step* | *lst*     | *acc*     |
|------+---------+---------|
| *1*    | (1 2 3) | NIL     |
| *2*    | (2 3)   | (1)     |
| *3*    | (3)     | (2 1)   |
| *4*    | NIL     | (3 2 1) |

Time Complexity: O(n)


*** Iterative (Loop Based)
#+begin_src lisp
  (defun my-reverse-loop (lst)
    (let ((result nil))
      (dolist (x lst result)
        (setf result (cons x result)))))

  (my-reverse-loop '(1 2 3))
#+end_src


*** Destructive Reverse (Advanced)
#+begin_src lisp
  (defun my-nreverse (lst)
    (let ((prev nil)
  	(curr lst))
      (loop while curr do
        (let ((next (cdr curr)))
  	(setf (cdr curr) prev)
  	(setf prev curr)
  	(setf curr next)))
      prev))

  (my-nreverse '(1 2 3))
#+end_src
- Original list modify karta hai
- Fastest hai (no extra memory)



** Min-Max

*** Iterative Approach
#+begin_src lisp
(defun min-max (lst)
  (if (null lst)
      nil
      (let ((min (car lst))
	    (max (car lst)))
	(dolist (x (cdr lst) (list min max))
	  (when (< x min)
	    (setf min x))
	  (when (> x max)
	    (setf max x))))))

(min-max '(5 2 8 1 9 3))
#+end_src

*** Recursive Approach
#+begin_src lisp
(defun min-max-rec (lst min max)
  (if (null lst)
      (list min max)
      (let ((x (car lst)))
	(min-max-rec
	 (cdr lst)
	 (if (< x min) x min)
	 (if (> x max) x max)))))

(let ((lst '(5 2 8 1 9 3)))
  (min-max-rec (cdr lst) (car lst) (car lst)))
#+end_src
